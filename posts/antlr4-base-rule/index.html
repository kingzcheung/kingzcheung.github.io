ANTLR4 匹配基础数据类型常用词法规则

ANTLR4 匹配基础数据类型常用词法规则

匹配数字

INT: '0'..'9'+ ;
// 或者
INT2: [0-9]+;

匹配浮点数

FLOAT: DIGIT+'.' DIGIT* //匹配1.39 3.1415926
    | '.'DIGIT+ //匹配 .1 或者 .1415926
    ;
fragment
DIGIT: [0-9];

匹配字符串常量

STRING: '"' (ESC|.)*? '"';

fragment
ESC: '\\"' | '\\\\'; //双字符序列 \" 和 \\

匹配注释

LINE_COMMENT:'//' .*?'\r'? '\n'  -> skip;
COMMENT: '/*' .*? '*/' -> skip;

匹配空白字符

WS: (' '|'\t'|'\r'|'\n')+ ->skip;
//or
WS2: [ \t\r\n]+ ->skip;

反向丢弃

STUFF: ~'\n'+ ->skip; // 除了'\n' 之外的字符全部丢弃
